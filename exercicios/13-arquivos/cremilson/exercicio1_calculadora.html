// Interface Operacao
class Soma {
  calcular(a, b) { return a + b; }
}
class Subtracao {
  calcular(a, b) { return a - b; }
}
class Multiplicacao {
  calcular(a, b) { return a * b; }
}
class Potencia {
  calcular(a, b) { return a ** b; }
}
class DivisaoSegura {
  calcular(a, b) {
    if (b === 0) throw new Error("Divisão por zero!");
    return a / b;
  }
}

function executarOperacao(a, b, operacao) {
  try {
    return operacao.calcular(a, b);
  } catch (e) {
    return `Erro: ${e.message}`;
  }
}
