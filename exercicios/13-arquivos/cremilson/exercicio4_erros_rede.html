// Função async com try/catch para tratar erros
async function buscarJSON(fonte) {
  try {
    const res = await fetch(fonte.url);
    if (!res.ok) {
      throw new Error(`HTTP ${res.status}: ${res.statusText}`);
    }
    const dados = await res.json();
    exibirDados(dados);
  } catch (e) {
    exibirErro(`Erro de rede: ${e.message}`);
  }
}
